internal class Program
{
    private static void Main(string[] args)
    {
        Console.Write("Ingrese ángulo en grados: ");
        double angulo = Convert.ToDouble(Console.ReadLine());

        double rad = angulo * Math.PI / 180;

        Console.WriteLine("1. Seno\n2. Coseno\n3. Tangente");
        int opcion = Convert.ToInt32(Console.ReadLine());

        switch (opcion)
        {
            case 1:
                Console.WriteLine("Seno: " + Math.Sin(rad));
                break;
            case 2:
                Console.WriteLine("Coseno: " + Math.Cos(rad));
                break;
            case 3:
                Console.WriteLine("Tangente: " + Math.Tan(rad));
                break;
            default:
                Console.WriteLine("Opción inválida");
                break;
        }
    }
}
